```typescript
import * as fs from 'fs';
import * as path from 'path';
import { execSync } from 'child_process';
import { detectGemini } from '../platforms/gemini';
import { detectQwen } from '../platforms/qwen';

export interface PlatformItem {
  label: string;
  value: string;
  hint?: string;
  disabled?: boolean;
}

export function detectProjectName(): string {
  try {
    const result = execSync('git rev-parse --show-toplevel', {
      encoding: 'utf-8',
      stdio: 'pipe',
    }).trim();
    return path.basename(result);
  } catch {
    return path.basename(process.cwd());
  }
}

export function detectPlatforms(): PlatformItem[] {
  const home = process.env.HOME || process.env.USERPROFILE || '';
  const isInPath = (cmd: string): boolean => {
    try {
      execSync(`which ${cmd}`, { stdio: 'pipe' });
      return true;
    } catch {
      return false;
    }
  };

  return [
    {
      label: 'Claude Code',
      value: 'claude-code',
      hint: fs.existsSync(path.join(home, '.claude', 'settings.json')) ? '✓ detected' : 'not found',
    },
    {
      label: 'Codex',
      value: 'codex',
      hint: isInPath('codex') ? '✓ detected' : 'not found',
    },
    {
      label: 'Gemini CLI',
      value: 'gemini',
      hint: detectGemini() ? '✓ detected' : 'not found',
    },
    {
      label: 'Qwen Code',
      value: 'qwen',
      hint: detectQwen() ? '✓ detected' : 'not found',
    },
    {
      label: 'Cursor',
      value: 'cursor',
      hint: fs.existsSync(path.join(process.cwd(), '.cursor')) ? '✓ detected' : 'not found',
    },
  ];
}
```

```typescript
import { findGitRoot } from '../core/store';
import { initConfig } from '../config';
import { detectProjectName, detectPlatforms } from '../core/platform-detector';
import { installClaudeCode } from '../platforms/claude-code';
import { installCursor } from '../platforms/cursor';
import { installCodex } from '../platforms/codex';
import { installGemini } from '../platforms/gemini';
import { installQwen } from '../platforms/qwen';

export interface QuickInitOptions {
  platform?: string;
}

export async function quickInit(options: QuickInitOptions): Promise<void> {
  const cwd = process.cwd();
  const repoRoot = findGitRoot(cwd) ?? cwd;
  const projectName = detectProjectName();

  initConfig(repoRoot, projectName);

  const allPlatforms = detectPlatforms();
  const targets = options.platform
    ? options.platform.split(',').map((s) => s.trim())
    : allPlatforms.filter((p) => p.hint?.includes('✓')).map((p) => p.value);

  const installed: string[] = [];
  for (const p of targets) {
    if (p === 'claude-code') {
      await installClaudeCode(repoRoot);
      installed.push(p);
    } else if (p === 'cursor') {
      await installCursor(repoRoot);
      installed.push(p);
    } else if (p === 'codex') {
      await installCodex(repoRoot);
      installed.push(p);
    } else if (p === 'gemini') {
      await installGemini(repoRoot);
      installed.push(p);
    } else if (p === 'qwen') {
      await installQwen(repoRoot);
      installed.push(p);
    }
  }

  const platformList = installed.length ? installed.join(', ') : 'none';
  console.log(`✓ memobank initialized (project: ${projectName}, platforms: ${platformList})`);
  if (!installed.length) {
    console.log('  Tip: run memo init --interactive to configure platforms manually.');
  }
}
```

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as readline from 'readline';
import matter from 'gray-matter';
import { findRepoRoot } from '../core/store';

export interface StudyOptions {
  if?: string;
  list?: boolean;
  repo?: string;
}

function findLessons(repoRoot: string): string[] {
  const dirs = ['lesson', 'decision', 'workflow', 'architecture'];
  const results: string[] = [];
  for (const dir of dirs) {
    const dirPath = path.join(repoRoot, '.memobank', dir);
    if (!fs.existsSync(dirPath)) continue;
    for (const file of fs.readdirSync(dirPath)) {
      if (file.endsWith('.md')) results.push(path.join(dirPath, file));
    }
  }
  return results;
}

function findLesson(repoRoot: string, name: string): string | null {
  const files = findLessons(repoRoot);
  // Exact slug match first
  const exact = files.find((f) => path.basename(f, '.md') === name);
  if (exact) return exact;
  // Fuzzy: file contains the name
  return files.find((f) => path.basename(f, '.md').includes(name)) ?? null;
}

function findClaudeMd(repoRoot: string): string {
  const candidate = path.join(repoRoot, 'CLAUDE.md');
  if (fs.existsSync(candidate)) return candidate;
  return candidate; // create at git root if missing
}

function buildBlock(condition: string, lessonPath: string, summary: string): string {
  const relPath = path.relative(process.cwd(), lessonPath);
  return `\n<important if="${condition}">\n<!-- source: ${relPath} -->\n${summary.trim()}\n</important>\n`;
}

function extractSummary(content: string): string {
  // Take first 4 non-empty lines of body
  const lines = content.split('\n').filter((l) => l.trim() !== '');
  return lines.slice(0, 4).join('\n');
}

async function promptCondition(lessonName: string): Promise<string> {
  const rl = readline.createInterface({ input: process.stdin, output: process.stdout });
  return new Promise((resolve) => {
    rl.question(
      `Condition for "${lessonName}" (e.g. "you are installing dependencies"): `,
      (answer) => {
        rl.close();
        resolve(answer.trim());
      }
    );
  });
}

export async function studyCommand(
  lessonName: string | undefined,
  options: StudyOptions
): Promise<void> {
  const repoRoot = findRepoRoot(process.cwd(), options.repo);

  if (options.list || !lessonName) {
    const files = findLessons(repoRoot);
    if (files.length === 0) {
      console.log('No lessons found in .memobank/');
      return;
    }
    console.log('Available lessons:');
    for (const f of files) {
      const rel = path.relative(repoRoot, f);
      const parsed = matter(fs.readFileSync(f, 'utf-8'));
      console.log(`  ${path.basename(f, '.md')}  —  ${parsed.data.description ?? rel}`);
    }
    return;
  }

  const lessonPath = findLesson(repoRoot, lessonName);
  if (!lessonPath) {
    throw new Error(
      `Lesson not found: "${lessonName}". Run memo study --list to see available lessons.`
    );
  }

  const parsed = matter(fs.readFileSync(lessonPath, 'utf-8'));
  const summary = extractSummary(parsed.content);

  const claudeMdPath = findClaudeMd(repoRoot);
  const existing = fs.existsSync(claudeMdPath) ? fs.readFileSync(claudeMdPath, 'utf-8') : '';

  // Check if already studied
  const relPath = path.relative(process.cwd(), lessonPath);
  if (existing.includes(`<!-- source: ${relPath} -->`)) {
    console.warn(`⚠  Already studied: "${lessonName}" is already in CLAUDE.md`);
    return;
  }

  const condition = options.if ?? (await promptCondition(lessonName));
  if (!condition) {
    throw new Error('Condition is required. Use --if="..." or enter it interactively.');
  }

  const block = buildBlock(condition, lessonPath, summary);
  fs.appendFileSync(claudeMdPath, block, 'utf-8');
  console.log(`✓ Lesson "${lessonName}" studied → CLAUDE.md updated`);
}
```

```typescript
const GITIGNORE_ENTRIES: Array<{ entry: string; comment: string }> = [
  { entry: '.memobank/meta/access-log.json', comment: 'access log is local, not team state' },
  { entry: '.memobank/meta/code-index.db', comment: 'code symbol index — binary, not committed' },
  { entry: '.memobank/.lancedb/', comment: 'vector index — binary, not committed' },
  { entry: '.memobank/pending/', comment: 'capture queue — transient, not committed' },
];

function ensureGitignore(repoRoot: string): void {
  // repoRoot is the .memobank/ dir; git root is one level up
  const gitRoot = path.dirname(repoRoot);
  const gitignorePath = path.join(gitRoot, '.gitignore');

  let content = fs.existsSync(gitignorePath) ? fs.readFileSync(gitignorePath, 'utf-8') : '';

  const toAdd = GITIGNORE_ENTRIES.filter(({ entry }) => !content.includes(entry));
  if (!toAdd.length) return;

  const block = '\n# memobank\n' + toAdd.map(({ entry }) => entry).join('\n') + '\n';
  if (!content) {
    fs.writeFileSync(gitignorePath, block.trimStart());
  } else {
    fs.appendFileSync(gitignorePath, block);
  }
}
```
